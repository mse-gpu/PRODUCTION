#version 0.0.1
#Exemple de paramètre pour la création d'une jvm

# Memory
-Xms100m -Xmx500m	# HeapMin HeapMax , m=mega, g=giga
-Xss5M 				# Stack: defaut 2m. Conseil: Ne pas changer

# verbose
-verbose:jni
-verbose:gc
-verbose

# Optimiser:
-server 				# Installation: copier coller le rep bin/serveur du jre du jdk dans le jre pur
-XX:+UseParallelGC 		# perform gctime, but bloc application
-XX:+UseConcMarkSweepGC	# minimise gcpause

# Path
-Djava.library.path="XXX" # exemple: XXX=C:\Program Files\Java\externe\bin

#assertion
-disablesystemassertions 

# Rendu GL
-Dsun.java2d.opengl=true
-Dsun.java2d.opengl=True 		# Attention:  T majuscule affiche dans la console la disponibilite
-Dsun.java2d.accthreshold=0

# Rendu non GL
-Dsun.java2d.translaccel=true
-Dsun.java2d.ddforcevram=true
-Dsun.java2d.ddscale=true
-Dsun.java2d.noddraw=true 		#  desactive directdraw. Objectif: Eliminer le scintillement, contourner bug old jvm

# Profiler:
-Dcom.sun.management.jmxremote 

#Divers
-Dfile.encoding=UTF-8

# Mode emploi: 
Ouvrir une autre console, puis
     jconsole idProcess <cr>, 
idProcess peut être vu dans taskManager, ou
 	jconsole <cr> 
puis demander une connection vers l'une ou l'autre des JVM par click dans le GUI
